### 35. Search Insert Position
* `Binary Search`
* O(lg(n)), O(1)
* LI :
* C++
```cpp
class Solution {
public:
    int searchInsert(vector<int>& a, int t) {
        int l = 0, r = a.size() - 1;
        
        while(l <= r){
            int m = (l + r) / 2;
            
            if(t == a.at(m)){
                return m;
            }
            
            if(t < a.at(m)){
                r = m - 1;
            }else{
                l = m + 1;
            }
        }
        
        return l;
    }
};
```
