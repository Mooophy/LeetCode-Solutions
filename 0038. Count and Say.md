### 38. Count and Say
* `Lookahead`
* O(k * n), O(k) where k is the length of result each step
* C++
```cpp
class Solution {
public:
    string countAndSay(int n) {
        if(n == 1){
            return "1";
        }
        
        auto s = countAndSay(n - 1);
        string res;
        
        for(int i = 0, n = s.size(); i < n; ){
            int j = i + 1;
            for(; j < n && s.at(i) == s.at(j); ++j);
            res += to_string(j - i) + s.at(i);
            i = j;
        }
        
        return res;
    }
};
```
